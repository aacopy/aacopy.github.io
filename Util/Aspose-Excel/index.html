ExcelChunk

/**
 * excel注解
 * 用于在excel中规划一块区域，来存放一类字段
 * @author iseven.yang
 * @date 2021/12/15 13:14
 */
@Target({ElementType.TYPE})
@Retention(RetentionPolicy.RUNTIME)
public @interface ExcelChunk {

    /**
     * 一块区域的唯一标识，多个字段上标识的name相同，标识放在同一块区域
     */
    String name() default "";
    /**
     * 相对块的name，如果为空，则表示参照的基本块，一个sheet中只能有一个块的dependChunk为空
     */
    String dependChunk() default "";
    /**
     * 相对位置，四个位置，用，top，down，left， right来表示上下左右，默认在相对块的下面
     */
    ShipEnum position() default ShipEnum.DOWN;
    /**
     * 行偏移量
     */
    int offsetX() default 0;
    /**
     * 列偏移量
     */
    int offsetY() default 0;
    /**
     * 一块区域中固定部分的单元格
     */
    ExcelValue[] fixCells() default @ExcelValue;
}

ExcelEntity

/**
 * excel注解
 * 标识某个类，设置excel中的属性
 * @author iseven.yang
 * @date 2021/12/15 14:13
 */
@Target({ElementType.TYPE, ElementType.FIELD})
@Retention(RetentionPolicy.RUNTIME)
public @interface ExcelEntity {
    /**
     * 生成excel的文件名
     */
    String fileName() default "";

    /**
     * sheet名称
     */
    String sheetName() default "";

    /**
     * 列宽，从第一列宽度开始写
     * @return
     */
    double[] cellsWidth() default 8.43;

    /**
     * 行高
     * @return
     */
    double rowHeight() default 25;

    /**
     * 单元格锁定
     * @return
     */
    LockModeEnum lock() default LockModeEnum.NONE;

    /**
     * 全局不锁定，只锁定传入的单元格，（英文逗号，中间不要带空格）
     * String格式
     * 如果为整行参数格式：R,rowNum,startCell （行索引，开始列）
     * 如果为整列参数格式：C,cellNum,startRow （列索引，开始行）
     * 如果取范围内的cell，参数格式：beginRow,beginCell,rowNum,cellNum （开始行,开始列,总行数,总列数）
     * @return
     */
    String[] lockInclude() default "";

    /**
     * 全局锁定，排除传入的单元格，（英文逗号，中间不要带空格）
     * String格式
     * 如果为整行参数格式：rowNum,startCell （行索引，开始列）
     * 如果为整列参数格式：cellNum,startRow （列索引，开始行）
     * 如果取范围内的cell，参数格式：beginRow,beginCell,rowNum,cellNum （开始行,开始列,总行数,总列数）
     * @return
     */
    String[] lockExclude() default "";

    /**
     * 划分块，主要用来解决存在动态行或列的情况下定位单元格位置
     */
    ExcelChunk[] chunks() default @ExcelChunk;
}

ExcelValue

/**
 * @author iseven.yang
 * @date 2021/12/15 13:59
 */
@Target({ElementType.FIELD})
@Retention(RetentionPolicy.RUNTIME)
public @interface ExcelValue {
    /**
     * 所属区域，在字段上的注解，需要设置该值，绑定区域；在类上的注解不需要设置该值
     */
    String chunkName() default "";
    /**
     * 起始位置行坐标x，x为相对于块（ExcelChunk）中的位置，不是excel中的绝对位置
     */
    int x() default 0;
    /**
     * 起始位置列坐标y，y为相对于块（ExcelChunk）中的位置，不是excel中的绝对位置
     */
    int y() default 0;
    /**
     * 如果有单元格合并，需要跨的行数
     */
    int xNum() default 1;
    /**
     * 如果有单元格合并，需要跨的列数
     */
    int yNum() default 1;
    /**
     * 单元格样式
     */
    StyleEnum style() default StyleEnum.COMMON;
    /**
     * 单元格内容，在字段上的注解：填写对象属性中需要获取的值
     * 在类上的注解需要设置该值，用于设置单元格中固定的数据
     */
    String[] value() default "";
    /**
     * 如果字段使一个列表，需要指定列表中的数据填充方向，允许的值：down，right。默认往下
     */
    DirectionEnum direction() default DirectionEnum.DOWN;

    /**
     * 获取值的方式，value表示直接获取，obj，表示需要从对象内部属性获取，嵌套获取
     * @return
     */
    ValueTypeEnum type() default ValueTypeEnum.VALUE;

    /**
     * 自定义处理方法，入参为静态方法名，无返回值，方法入参固定两个（Object value, Worksheet worksheet, Cell cell）
     * @return
     */
    String handler() default "";
}

package cn.aacopy.test.aspose.excel.util;

import cn.aacopy.test.aspose.excel.enums.StyleEnum;
import com.aspose.cells.*;

/**
 * @author iseven.yang
 * @date 2021/12/16 19:35
 */
public class ExcelStyle {

    /**
     * 初始化所有样式
     * @param workbook
     */
    public static void initAllStyle(Workbook workbook) {
        StyleEnum[] styles = StyleEnum.values();
        initStyles(workbook, styles);
    }

    /**
     * 初始化选的的样式
     * @param workbook
     * @param styles
     */
    public static void initStyles(Workbook workbook, StyleEnum... styles) {
        if(styles == null) return;
        for(StyleEnum style: styles) {
            switch (style) {
                case COMMON: commonStyle(workbook); break;
                case TITLE: titleStyle(workbook); break;
                case MODULE_TITLE: moduleTitleStyle(workbook); break;
                case KEY: keyStyle(workbook); break;
            }
        }
    }

    /**
     * 基础样式
     * @param style
     */
    public static void baseStyle(Style style) {
        //字体
        Font font = style.getFont();
        font.setName("微软雅黑");
        font.setSize(12);
        //自动换行
        style.setTextWrapped(true);
        //居中显示
        style.setVerticalAlignment(TextAlignmentType.CENTER);
        style.setHorizontalAlignment(TextAlignmentType.CENTER);
        //白底背景色
        style.setForegroundColor(Color.fromArgb(255, 255, 255));
        style.setPattern(BackgroundType.SOLID);
        //全边框
        style.setBorder(BorderType.TOP_BORDER, CellBorderType.THIN, Color.getBlack());
        style.setBorder(BorderType.BOTTOM_BORDER, CellBorderType.THIN, Color.getBlack());
        style.setBorder(BorderType.LEFT_BORDER, CellBorderType.THIN, Color.getBlack());
        style.setBorder(BorderType.RIGHT_BORDER, CellBorderType.THIN, Color.getBlack());
    }

    /**
     * 通用样式
     * @param workbook
     */
    public static void commonStyle(Workbook workbook) {
        Style style = workbook.createStyle();
        style.setName(StyleEnum.COMMON.name());
        baseStyle(style);
    }

    /**
     * 创建标题样式
     * @param workbook
     */
    public static void titleStyle(Workbook workbook) {
        Style style = workbook.createStyle();
        style.setName(StyleEnum.TITLE.name());
        baseStyle(style);
        Font font = style.getFont();
        //字体大小
        font.setSize(18);
        //加粗
        font.setBold(true);
        //字体红色
        font.setColor(Color.fromArgb(192, 0, 0));
        //灰色背景
        style.setForegroundColor(Color.fromArgb(242, 242, 242));
    }

    /**
     * 模块标题样式
     * @param workbook
     */
    public static void moduleTitleStyle(Workbook workbook) {
        Style style = workbook.createStyle();
        style.setName(StyleEnum.MODULE_TITLE.name());
        baseStyle(style);
        Font font = style.getFont();
        //加粗
        font.setBold(true);
        //灰色背景
        style.setForegroundColor(Color.fromArgb(242, 242, 242));
        style.setHorizontalAlignment(TextAlignmentType.LEFT);
    }

    /**
     * key样式
     * @param workbook
     */
    public static void keyStyle(Workbook workbook) {
        Style style = workbook.createStyle();
        style.setName(StyleEnum.KEY.name());
        baseStyle(style);
        Font font = style.getFont();
        //字体灰色
        font.setColor(Color.fromArgb(128, 128, 128));
    }

}

    /**
     * 导入转二维数组
     * @param inputStream
     * @return
     */
    public static Object[][] importData(InputStream inputStream) throws Exception {
        Workbook workbook = new Workbook(inputStream);
        Worksheet worksheet = workbook.getWorksheets().get(0);
        Cells cells = worksheet.getCells();
        Object[][] dataTable = worksheet.getCells().exportArray(0, 0, cells.getMaxRow()+1, cells.getMaxColumn()+1);
        return dataTable;
    }

    /**
     * 导出
     * @param obj
     * @throws Exception
     */
    public static void export(Object obj) throws Exception {
        List<Object> sheetList = new ArrayList<>();
        findAllSheet(sheetList, obj);
        //创建excel
        Workbook workbook = new Workbook();
        //初始化样式
        ExcelStyle.initAllStyle(workbook);
        ExcelStyle.initAllStyle(workbook);
        WorksheetCollection worksheets = workbook.getWorksheets();
        String fileName = obj.getClass().getSimpleName() + ".xlsx";
        if(sheetList.size()>0) {
            for(int i=0; i<sheetList.size(); i++) {
                if(i>0) {
                    worksheets.add();
                }
                Worksheet sheet = worksheets.get(i);
                Object o = sheetList.get(i);
                //生成sheet
                createSheet(sheet, o, obj.getClass());
            }
        }
        workbook.save("E:\\" + fileName);
    }

    /**
     * 递归获取所有需要生成sheet的对象，放在sheetList集合中,如果循环依赖，就直接抛出异常，调用方自己处理
     * @param sheetList
     * @param obj
     * @throws IllegalAccessException
     */
    public static void findAllSheet(List<Object> sheetList, Object obj) throws IllegalAccessException {
        if(obj == null) {
            return;
        }
        Class<?> aClass = obj.getClass();
        ExcelEntity excelEntityAnno = aClass.getAnnotation(ExcelEntity.class);
        if(excelEntityAnno != null) {
            sheetList.add(obj);
        }
        Field[] fields = aClass.getDeclaredFields();
        if(fields != null) {
            for(Field field: fields) {
                if(field.getAnnotation(ExcelEntity.class) != null) {
                    field.setAccessible(true);
                    findAllSheet(sheetList, field.get(obj));
                }
            }
        }
    }

    /**
     * 创建sheet
     * @param sheet
     * @param obj
     * @throws Exception
     */
    public static void createSheet(Worksheet sheet, Object obj, Class rootClass) throws Exception {
        Class<?> aClass = obj.getClass();
        ExcelEntity excelEntityAnno = aClass.getAnnotation(ExcelEntity.class);
        if(excelEntityAnno == null) {
            //如果类上没有标识时一个excel导出类，就在字段上找@ExcelSheet注解
            log.warn("类未添加ExcelEntity注解");
            return;
        }

        //设置默认sheet名
        String sheetName = excelEntityAnno.sheetName();
        if(StringUtils.isBlank(sheetName)) {
            sheetName = aClass.getSimpleName();
        }

        sheet.setName(sheetName);
        Cells cells = sheet.getCells();

        //设置列宽
        double[] widths = excelEntityAnno.cellsWidth();
        for (int i = 0; i < widths.length; i++) {
            cells.setColumnWidth(i, widths[i]);
        }

        //通过两个坐标（x,y）和（m,n）确定每个块的边界，Integer数字长度为4，0：x, 1:y, 2:m, 3:n
        Map<String, Integer[]> chunkArea = new HashMap<>();
        //获取chunk
        ExcelChunk[] chunkAnnos = excelEntityAnno.chunks();
        //按照chunk依赖关系分组，同一个依赖放一组
        Map<String, List<ExcelChunk>> dependGroup = Arrays.stream(chunkAnnos).collect(Collectors.groupingBy(excelChunk -> excelChunk.dependChunk()));
        //获取字段数据
        List<Map<String, Object>> allValueField = getAllValueField(obj);
        //需要按照chunk名称分组，同一个chunk的放一组，方便按依赖顺序生成，如果存在多层级需要递归出所有的分组数据
        Map<String, List<Map<String, Object>>> filedGroup = allValueField.stream()
                .collect(Collectors.groupingBy(fieldMap -> ((ExcelValue) fieldMap.get(KEY_EXCEL_VALUE)).chunkName()));

        //从依赖关系根节点往下生成，可以避免循环依赖问题
        List<ExcelChunk> rootList = dependGroup.get("");
        if(rootList.size() != 1) {
            throw new RuntimeException("根chunk错误");
        }
        //设置根chunk数据
        ExcelChunk rootChunk = rootList.get(0);
        Integer[] rootArea = new Integer[]{rootChunk.offsetX(), rootChunk.offsetY(), rootChunk.offsetY(), rootChunk.offsetY()};
        chunkArea.put(rootChunk.name(), rootArea);
        //处理固定数据
        setFixData(rootArea, rootChunk, sheet, rootClass);
        //处理字段数据
        List<Map<String, Object>> rootChunkFields = filedGroup.get(rootChunk.name());
        setFieldData(rootArea, rootChunkFields, sheet, rootClass);

        //设置依赖根节点后面的节点数据
        List<String> temp0 = new ArrayList<>();//用于存放下一级需要遍历的数据，避免使用递归
        temp0.add(rootChunk.name());
        List<String> temp1 = new ArrayList<>();
        List<String> temp = temp0;
        boolean flag = true;
        for(int i=0; i<temp.size(); i++) {
            String dependChunk = temp.get(i);
            if(StringUtils.isNotBlank(dependChunk)) {
                List<ExcelChunk> chunks = dependGroup.get(dependChunk);
                if (chunks != null) {
                    for (ExcelChunk e : chunks) {
                        String name = e.name();
                        if (flag) {
                            temp1.add(name);
                        } else {
                            temp0.add(name);
                        }
                        //处理该chunk的数据
                        //获取依赖的chunk，计算当前chunk的初始位置，并将边界放在chunkArea中
                        Integer[] area = initPosition(dependChunk, chunkArea, e);
                        //处理固定数据
                        setFixData(area, e, sheet, rootClass);
                        //处理字段数据
                        List<Map<String, Object>> chunkFields = filedGroup.get(name);
                        setFieldData(area, chunkFields, sheet, rootClass);
                    }
                }
            }
            //当执行到最后一条数据的时候，切换list
            if(i==temp.size()-1) {
                temp.clear();
                temp = flag?temp1:temp0;
                flag = !flag;
                i=-1;
            }
        }
        //quectel背景
        sheet.setBackgroundImage(Base64.getDecoder().decode(BACKGROUND_IMG));
        //找到整个sheet的最后行列边界
        int maxRow = 0;
        int maxCell = 0;
        for(Integer[] is: chunkArea.values()) {
            if(is[2] > maxRow) {
                maxRow = is[2];
            }
            if(is[3] > maxCell) {
                maxCell = is[3];
            }
        }
        //调整行高
        for(int i=rootArea[0]; i<=maxRow; i++) {
            //自适应行高
            sheet.autoFitRow(i);
            double rowHeight = cells.getRowHeight(i);
            cells.setRowHeight(i, Math.max(excelEntityAnno.rowHeight(), rowHeight));
            for(int j=rootArea[1]; j<=maxCell; j++) {
                Cell cell = cells.get(i, j);
                //将在数据区没有样式的cell补充上通用样式
                if(cell.getValue() == null) {
                    cell.setStyle(sheet.getWorkbook().getNamedStyle(StyleEnum.COMMON.name()));
                }
            }
        }

        //锁定单元格
        LockModeEnum lock = excelEntityAnno.lock();
        StyleFlag styleFlag = new StyleFlag();
        styleFlag.setLocked(true);
        if(LockModeEnum.ALL == lock) {
            sheet.protect(ProtectionType.ALL);
        } else if(LockModeEnum.EXCLUDE == lock) {
            String[] strings = excelEntityAnno.lockExclude();
            setLock(cells, strings, false, styleFlag);
            sheet.protect(ProtectionType.ALL);
        } else if(LockModeEnum.INCLUDE == lock) {
            String[] strings = excelEntityAnno.lockInclude();
            for (int i = 0; i <= 255; i++) {
                Style style = cells.getColumns().get(i).getStyle();
                style.setLocked(false);
                cells.getColumns().get(i).applyStyle(style, styleFlag);
            }
            setLock(cells, strings, true, styleFlag);
            sheet.protect(ProtectionType.ALL);
        }

    }

    /**
     * 获取所有的填充在excel中的字段
     * @param obj
     * @return map中两个key，field和dataObj
     */
    private static List<Map<String, Object>> getAllValueField(Object obj) throws IllegalAccessException {
        if(obj == null) {
            return null;
        }
        List<Map<String, Object>> result = new ArrayList<>();
        Field[] fields = obj.getClass().getDeclaredFields();
        for(Field field: fields) {
            ExcelValues excelValues = field.getAnnotation(ExcelValues.class);
            if(excelValues != null) {
                ExcelValue[] values = excelValues.values();
                for(ExcelValue excelValue: values) {
                    getExcelValueAnno(excelValue, field, obj, result);
                }
            }
            ExcelValue excelValue = field.getAnnotation(ExcelValue.class);
            getExcelValueAnno(excelValue, field, obj, result);
        }
        return result;
    }

    /**
     * 设置单个excelValue注解
     * @param excelValue
     * @param field
     * @param obj
     * @param result
     * @throws IllegalAccessException
     */
    private static void getExcelValueAnno(ExcelValue excelValue, Field field, Object obj, List<Map<String, Object>> result) throws IllegalAccessException {
        if(excelValue != null) {
            //获取字段属性值
            field.setAccessible(true);
            Object value = field.get(obj);
            if(ValueTypeEnum.OBJ == excelValue.type()) {
                result.addAll(getAllValueField(value));
            } else if(ValueTypeEnum.VALUE == excelValue.type()) {
                Map<String, Object> map = new HashMap<>();
                map.put(KEY_FIELD, field);
                map.put(KEY_OBJ, value);
                map.put(KEY_EXCEL_VALUE, excelValue);
                result.add(map);
            }
        }
    }

    /**
     * 初始化块的位置,并将位置放在map中
     * @param dependChunk 依赖块的名称
     * @param chunkArea 块位置集合
     * @param e 当前块配置
     * @return
     */
    private static Integer[] initPosition(String dependChunk, Map<String, Integer[]> chunkArea, ExcelChunk e) {
        Integer[] integers = chunkArea.get(dependChunk);
        int beginX;
        int beginY;
        if(ShipEnum.RIGHT == e.position()) {//相对位置在右边
            beginX = integers[0] + e.offsetX();
            beginY = integers[3] + 1 + e.offsetY();
        } else { //默认相对位置在下面
            beginX = integers[2] + 1 + e.offsetX();
            beginY = integers[1] + e.offsetY();
        }
        Integer[] result = new Integer[4];
        result[0] = beginX;
        result[1] = beginY;
        result[2] = beginX;
        result[3] = beginY;
        //将数据放在map中
        chunkArea.put(e.name(), result);
        return result;
    }

    /**
     * 设置固定数据，并且修改块区域位置
     * @param area
     * @param e
     * @param sheet
     */
    private static void setFixData(Integer[] area, ExcelChunk e, Worksheet sheet, Class rootClazz) throws IllegalAccessException, InvocationTargetException, NoSuchMethodException {
        //获取固定单元格数据
        ExcelValue[] excelFixCellAnnos = e.fixCells();
        for(ExcelValue excelFixCell: excelFixCellAnnos) {
            setExcelValue(area, excelFixCell.value(), excelFixCell, sheet, rootClazz);
        }
    }

    /**
     * 设置字段数据
     * @param area 块区域数据
     * @param chunkFields 字段
     * @param sheet
     * @throws IllegalAccessException
     */
    private static void setFieldData(Integer[] area, List<Map<String, Object>> chunkFields, Worksheet sheet, Class rootClazz) throws IllegalAccessException, NoSuchFieldException, InstantiationException, InvocationTargetException, NoSuchMethodException {
        if(chunkFields != null && chunkFields.size()>0) {
            for(Map<String, Object> fieldMap: chunkFields) {
                Field field = (Field) fieldMap.get(KEY_FIELD);
                ExcelValue excelValueAnno = (ExcelValue) fieldMap.get(KEY_EXCEL_VALUE);
                //获取字段属性值
                field.setAccessible(true);
                Object value = fieldMap.get(KEY_OBJ);
                //如果是多层级取值
                String[] valueKeys = excelValueAnno.value();
                if(valueKeys.length == 1 ) {
                    if(StringUtils.isNotBlank(valueKeys[0])) {
                        value = getFiledValue(value, valueKeys[0].split("\\."), 0);
                    }
                } else if(valueKeys.length > 1) {
                    List values = new ArrayList();
                    for(String vk: valueKeys) {
                        if(StringUtils.isNotBlank(vk)) {
                            values.add(getFiledValue(value, vk.split("\\."), 0));
                        } else {
                            values.add(vk);
                        }
                    }
                    value = values;
                }
                setExcelValue(area, value, excelValueAnno, sheet, rootClazz);
            }
        }
    }

    /**
     * 递归获取字段的数据
     * @param dataObj
     * @param keys
     * @param index
     * @return
     * @throws IllegalAccessException
     * @throws NoSuchFieldException
     */
    private static Object getFiledValue(Object dataObj, String[] keys, int index) throws IllegalAccessException, NoSuchFieldException {
        if(dataObj instanceof Map) {
            Object o = ((Map) dataObj).get(keys[index]);
            if(keys.length > index+1 && o != null) { //如果还需要取下级
                return getFiledValue(o, keys, index + 1);
            } else {
                return o;
            }
        } else if(dataObj instanceof Object[] || dataObj instanceof Collection) { //如果是集合，需要取集合内的数据
            if(dataObj instanceof Collection) {
                dataObj = ((Collection) dataObj).toArray();
            }
            List result = new ArrayList();
            for(Object o: (Object[]) dataObj) { //循环获取集合中的数据
                if(keys.length > index+1 && o != null) { //如果还需要取下级
                    result.add(getFiledValue(o, keys, index+1)); //递归获取数据
                } else {
                    Field field = o.getClass().getDeclaredField(keys[index]);
                    //获取字段属性值
                    field.setAccessible(true);
                    Object value = field.get(o);
                    result.add(value);
                }
            }
            return result;
        } else { //如果是对象
            Field field = dataObj.getClass().getDeclaredField(keys[index]);
            //获取字段属性值
            field.setAccessible(true);
            Object value = field.get(dataObj);
            if(keys.length > index+1 && value != null) { //如果还需要取下级
                return getFiledValue(value, keys, index + 1);
            } else {
                return value;
            }
        }
    }

    /**
     * 在excel中设置数据
     * @param area 区域
     * @param value 数据
     * @param excelValueAnno 配置
     * @param sheet excel
     */
    private static void setExcelValue(Integer[] area, Object value, ExcelValue excelValueAnno, Worksheet sheet, Class rootClazz) throws IllegalAccessException, InvocationTargetException, NoSuchMethodException {
        //获取数据填充的起始位置
        int x = excelValueAnno.x();
        int y = excelValueAnno.y();
        int realX = area[0] + x; //excel中的真实坐标
        int realY = area[1] + y; //excel中的真实坐标
        int stepX = excelValueAnno.xNum(); //添加单个数据时的X步进
        int stepY = excelValueAnno.yNum(); //添加单个数据时的Y步进
        //自定义处理方法
        Method handlerMethod = null;
        String handler = excelValueAnno.handler();
        if(StringUtils.isNotBlank(handler)) {
            handlerMethod = rootClazz.getMethod(handler, Object.class, Worksheet.class, Cell.class);
        }
        if(value instanceof Collection || value instanceof String[]) {
            if(value instanceof Collection) {
                value = ((Collection) value).toArray();
            }
            Object[] values = (Object[]) value;
            if(values != null && values.length>0) {
                //获取填充方向
                DirectionEnum direction = excelValueAnno.direction();
                if(direction == DirectionEnum.RIGHT) { //向右填充
                    for(int j=0; j<values.length; j++) {
                        if(j>0) {
                            realY+=stepY;
                        }
                        setCellValue(sheet, excelValueAnno.style(), values[j], realX, realY, stepX, stepY, handlerMethod);
                    }
                    //计算边界位置
                    realY+=(stepY-1);
                    realX+=(stepX-1);
                } else { //默认向下填充
                    for(int j=0; j<values.length; j++) {
                        if(j>0) {
                            realX+=stepX;
                        }
                        setCellValue(sheet, excelValueAnno.style(), values[j], realX, realY, stepX, stepY, handlerMethod);
                    }
                    realX+=(stepX-1);
                    realY+=(stepY-1);
                }
            }
        } else {
            setCellValue(sheet, excelValueAnno.style(), value, realX, realY, stepX, stepY, handlerMethod);
            realX = realX+stepX-1;
            realY = realY+stepY-1;
        }

        //更新area最大位置记录
        if(realX > area[2]) {
            area[2] = realX;
        }
        if(realY > area[3]) {
            area[3] = realY;
        }
    }

    /**
     * 设置单元格
     * @param sheet
     * @param style
     * @param o
     * @param realX
     * @param realY
     * @param stepX
     * @param stepY
     */
    private static void setCellValue(Worksheet sheet, StyleEnum style, Object o, int realX, int realY, int stepX, int stepY, Method handler) throws InvocationTargetException, IllegalAccessException {
        Cells cells = sheet.getCells();
        Cell cell = cells.get(realX, realY);
        cell.setValue(o);
        cell.setStyle(sheet.getWorkbook().getNamedStyle(style.name()));
        //自定义处理方法执行
        if(handler != null) {
            //执行自定义代码静态方法
            handler.invoke(null, o, sheet, cell);
        } else {
            if(stepX>1||stepY>1){
                try {
                    cells.merge(realX, realY, stepX, stepY);
                } catch (Exception e) {
                    log.warn("合并单元格冲突..." + e.getMessage());
                }
            }
        }
    }

    /**
     * 设置单元格锁
     * @param cells
     * @param strings
     * @param lockFlag
     * @param styleFlag
     */
    private static void setLock(Cells cells, String[] strings, boolean lockFlag, StyleFlag styleFlag) {
        Style style;
        for(String s: strings) {
            String[] split = s.split(",");
            if(split.length == 3) { //整行，或者整列
                if("R".equals(split[0])) { //整行
                    int row = Integer.valueOf(split[1]);
                    style = cells.getRows().get(row).getStyle();
                    style.setLocked(lockFlag);
                    cells.getRows().get(row).applyStyle(style, styleFlag);
                    for(int i=0; i<Integer.valueOf(split[2]); i++) {
                        Style style1 = cells.get(row, i).getStyle();
                        style1.setLocked(!lockFlag);
                        cells.get(row, i).setStyle(style1);
                    }
                } else if("C".equals(split[0])) { //整列
                    int cell = Integer.valueOf(split[1]);
                    style = cells.getColumns().get(cell).getStyle();
                    style.setLocked(lockFlag);
                    cells.getColumns().get(cell).applyStyle(style, styleFlag);
                    for(int i=0; i<Integer.valueOf(split[2]); i++) {
                        Style style1 = cells.get(i, cell).getStyle();
                        style1.setLocked(!lockFlag);
                        cells.get(i, cell).setStyle(style1);
                    }
                }
            } else if(split.length == 4) { //范围
                for(int i=Integer.valueOf(split[0]); i<=Integer.valueOf(split[2]); i++) {
                    for(int j=Integer.valueOf(split[1]); j<=Integer.valueOf(split[3]); j++) {
                        style = cells.get(i, j).getStyle();
                        style.setLocked(lockFlag);
                        cells.get(i, j).setStyle(style);
                    }
                }
            }
        }
    }
